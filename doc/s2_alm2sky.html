program s2_alm2sky

        ! Uses
    use s2_types_mod
    use s2_sky_mod

        ! Calls
    call parse_options
    call s2_sky_compute_map
    call s2_sky_free
    call s2_sky_io_fits_write

        ! Variables
    character (len=S2_STRING_LEN) :: filename_sky
    character (len=S2_STRING_LEN) :: filename_alm
    type (s2_sky) :: sky
    logical :: compute_map = .false.
    integer :: nside = 64

        ! Subroutines and functions
    subroutine parse_options ()

end program s2_alm2sky
Read a HEALPix alm fits file and write the read map to a full s2_sky fits file. Compute map if requested.

Usage:

  • [-help]: Display usage information.
  • [-alm filename_alm]: Name of map HEAPix alm fits file to read.
  • [-sky filename_sky]: Name of full s2_sky fits file to write.
  • [-compute_map compute_map]: Logical specifying whether to compute map.
  • [-nside]: HEALPix nside of map if to be computed.
